<acceptance_criteria>

## AC-1: Unreachable Destination Returns Error

```gherkin
Given a unit with movement=2 at Hex::from_offset(0, 0)
When apply_action(Move { destination: Hex::from_offset(0, 4) }) is called  (4 steps away)
Then the result is Err(ActionError::DestinationUnreachable)
And the unit's position is unchanged
```

## AC-2: Scripted Scenario Executes Without Error

```gherkin
Given a 5×5 board with grassland terrain on all hexes
And unit 1 (faction 0, movement=4, 30-damage sword) at offset (0,0)
And unit 2 (faction 1, 20 HP, default_defense=0) at offset (2,0)
When this action sequence is applied:
  1. Move unit 1 to offset (1,0)       -- advance into melee range
  2. Attack unit 2 with unit 1         -- 30 damage > 20 HP
  3. EndTurn
Then every apply_action call returns Ok(())
```

## AC-3: Winner Determined — One Survivor

```gherkin
Given the final GameState from AC-2
Then state.units.len() == 1
And state.units contains unit_id 1  (the attacker survived)
And state.units does NOT contain unit_id 2  (the defender is dead)
```

</acceptance_criteria>

    ── game_state.rs: DestinationUnreachable + pathfinding wiring ─

    Add to ActionError:
      DestinationUnreachable,

    Add import at top of game_state.rs:
      use crate::pathfinding::{find_path, get_zoc_hexes};

    In the Action::Move arm of apply_action, AFTER the existing bounds and
    occupancy checks, add pathfinding validation:

      // Pathfinding validation — only when unit has a movement budget
      if unit.moved {
          return Err(ActionError::UnitAlreadyMoved);
      }
      // ... existing bounds and occupancy checks ...

      // Wire find_path when movement budget is set (movement > 0)
      let movement = state.units[&unit_id].movement;
      if movement > 0 {
          let zoc = get_zoc_hexes(state, state.units[&unit_id].faction);
          let costs = state.units[&unit_id].movement_costs.clone();
          let start = state.positions[&unit_id];
          let reachable = find_path(
              &state.board,
              &costs,
              1,           // default_movement_cost: 1 for uncategorised terrain
              start,
              destination,
              movement,
              &zoc,
              false,       // is_skirmisher: not implemented yet
          );
          if reachable.is_none() {
              return Err(ActionError::DestinationUnreachable);
          }
      }

      state.positions.insert(unit_id, destination);
      state.units.get_mut(&unit_id).unwrap().moved = true;
      Ok(())

    Add a test to the existing #[cfg(test)] block in game_state.rs:

    test_move_unreachable_returns_error():
      let board = Board::new(10, 10);
      let mut state = GameState::new(board);
      let mut unit = Unit::new(1, "fighter", 30, 0);
      unit.movement = 2; // budget = 2 steps
      state.place_unit(unit, Hex::ORIGIN);
      // 4 steps away — exceeds budget
      let far = Hex::from_offset(0, 4);
      let result = apply_action(&mut state, Action::Move { unit_id: 1, destination: far });
      assert_eq!(result, Err(ActionError::DestinationUnreachable));
      assert_eq!(state.positions[&1], Hex::ORIGIN); // position unchanged

    - Removing the movement=0 bypass — it keeps existing tests working
    - Implementing Skirmisher as a Unit flag here — leave is_skirmisher=false

  <action>
    Create norrust_core/tests/simulation.rs.

    This is a Cargo integration test — it compiles against the public API of
    norrust_core (the rlib) without access to private items.

    Imports:
      use std::collections::HashMap;
      use norrust_core::board::Board;
      use norrust_core::game_state::{apply_action, Action, GameState};
      use norrust_core::hex::Hex;
      use norrust_core::schema::AttackDef;
      use norrust_core::unit::Unit;

    Write one test function: test_headless_match_scenario()

      #[test]
      fn test_headless_match_scenario() {
          // ── Setup: 5×5 board, all grassland ──────────────────────
          let mut board = Board::new(5, 5);
          for col in 0..5_i32 {
              for row in 0..5_i32 {
                  board.set_terrain(Hex::from_offset(col, row), "grassland");
              }
          }
          let mut state = GameState::new_seeded(board, 99);

          // ── Unit 1: Faction 0 fighter at (0,0) ───────────────────
          let sword = AttackDef {
              id: "sword".to_string(),
              name: "Sword".to_string(),
              damage: 30,
              strikes: 1,
              attack_type: "blade".to_string(),
              range: "melee".to_string(),
          };
          let mut unit1 = Unit::new(1, "fighter", 30, 0);
          unit1.movement = 4;
          unit1.movement_costs = {
              let mut m = HashMap::new();
              m.insert("grassland".to_string(), 1u32);
              m
          };
          unit1.attacks = vec![sword];

          // ── Unit 2: Faction 1 archer at (2,0) — 20 HP, 0% defense ─
          let mut unit2 = Unit::new(2, "archer", 20, 1);
          unit2.movement = 4;
          unit2.default_defense = 0; // always hit

          state.place_unit(unit1, Hex::from_offset(0, 0));
          state.place_unit(unit2, Hex::from_offset(2, 0));

          // ── Turn 1, Faction 0 ─────────────────────────────────────
          // Action 1: Move unit 1 to (1,0) — adjacent to unit 2
          apply_action(
              &mut state,
              Action::Move { unit_id: 1, destination: Hex::from_offset(1, 0) },
          )
          .expect("Move unit 1 to (1,0) should succeed");

          // Action 2: Attack unit 2 with unit 1 (30 dmg > 20 HP → dead)
          apply_action(
              &mut state,
              Action::Attack { attacker_id: 1, defender_id: 2 },
          )
          .expect("Attack on unit 2 should succeed");

          // Action 3: End turn
          apply_action(&mut state, Action::EndTurn).expect("EndTurn should succeed");

          // ── Verify final state ────────────────────────────────────
          assert_eq!(state.units.len(), 1, "only one unit should survive");
          assert!(state.units.contains_key(&1), "unit 1 (attacker) must survive");
          assert!(!state.units.contains_key(&2), "unit 2 (defender) must be dead");
          assert!(!state.positions.contains_key(&2), "unit 2 position must be cleared");
      }

    Avoid:
    - Writing multiple complex scenarios in one test — keep it focused on the
      single acceptance-criterion scenario
    - Using private types or internal modules — integration tests can only use
      items declared pub in norrust_core's public API
  </action>
