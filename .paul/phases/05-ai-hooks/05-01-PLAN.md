<objective>
## Goal
Add a JSON serialization layer to the Rust core: a `StateSnapshot` type that represents
the full game state as a plain data structure serializable to JSON, and an `ActionRequest`
type that deserializes a JSON string into an `Action`. Expose both via GDExtension bridge
methods (`get_state_json()` and `apply_action_json()`).

## Purpose
Phase 5's purpose is to open the game engine to external AI agents. The JSON layer is the
foundation: once GDScript can call `get_state_json()` and `apply_action_json()`, any
external client (Python script, socket relay, LLM tool-call) can observe and drive the
game without ever touching GDScript or the Redot UI.

## Output
- `norrust_core/src/snapshot.rs`: `StateSnapshot`, `UnitSnapshot`, `TileSnapshot`,
  `ActionRequest` — all with serde derives; `StateSnapshot::from_game_state()` helper.
- `norrust_core/Cargo.toml`: `serde_json = "1"` added.
- `norrust_core/src/lib.rs`: `pub mod snapshot;` added.
- `norrust_core/src/gdext_node.rs`:
  - `get_state_json() -> GString` — serializes current state to JSON
  - `apply_action_json(json: GString) -> i32` — deserializes and applies action; returns 0
    on success, -99 on parse error, or the standard negative ActionError code on failure.

    use serde::{Deserialize, Serialize};
    use crate::board::Board;
    use crate::game_state::{Action, GameState};
    use crate::hex::Hex;

    /// Flat representation of a single hex tile's terrain.
    #[derive(Debug, Serialize)]
    pub struct TileSnapshot {
        pub col: i32,
        pub row: i32,
        pub terrain_id: String,
    }

    /// Flat representation of a single runtime unit.
    #[derive(Debug, Serialize)]
    pub struct UnitSnapshot {
        pub id: u32,
        pub def_id: String,
        pub col: i32,
        pub row: i32,
        pub faction: u8,
        pub hp: u32,
        pub max_hp: u32,
        pub moved: bool,
        pub attacked: bool,
    }

    /// Complete serializable snapshot of a GameState for external consumers.
    #[derive(Debug, Serialize)]
    pub struct StateSnapshot {
        pub turn: u32,
        pub active_faction: u8,
        pub cols: u32,
        pub rows: u32,
        pub terrain: Vec<TileSnapshot>,
        pub units: Vec<UnitSnapshot>,
    }

    impl StateSnapshot {
        pub fn from_game_state(state: &GameState) -> Self {
            let cols = state.board.cols;
            let rows = state.board.rows;

            // Collect all non-empty terrain hexes
            let terrain = (0..cols as i32)
                .flat_map(|col| (0..rows as i32).map(move |row| (col, row)))
                .filter_map(|(col, row)| {
                    let hex = Hex::from_offset(col, row);
                    state.board.terrain_at(hex).map(|t| TileSnapshot {
                        col,
                        row,
                        terrain_id: t.to_owned(),
                    })
                })
                .collect();

            // Collect all units with positions
            let units = state
                .positions
                .iter()
                .filter_map(|(&uid, &hex)| {
                    let unit = state.units.get(&uid)?;
                    let (col, row) = hex.to_offset();
                    Some(UnitSnapshot {
                        id: uid,
                        def_id: unit.def_id.clone(),
                        col,
                        row,
                        faction: unit.faction,
                        hp: unit.hp,
                        max_hp: unit.max_hp,
                        moved: unit.moved,
                        attacked: unit.attacked,
                    })
                })
                .collect();

            StateSnapshot { turn, active_faction: state.active_faction, cols, rows, terrain, units }
        }
    }

    /// JSON-deserializable action request from external clients.
    ///
    /// Expected formats:
    ///   {"action":"Move","unit_id":1,"col":3,"row":2}
    ///   {"action":"Attack","attacker_id":1,"defender_id":2}
    ///   {"action":"EndTurn"}
    #[derive(Debug, Deserialize)]
    #[serde(tag = "action")]
    pub enum ActionRequest {
        Move { unit_id: u32, col: i32, row: i32 },
        Attack { attacker_id: u32, defender_id: u32 },
        EndTurn,
    }

    impl From<ActionRequest> for Action {
        fn from(req: ActionRequest) -> Self {
            match req {
                ActionRequest::Move { unit_id, col, row } => Action::Move {
                    unit_id,
                    destination: Hex::from_offset(col, row),
                },
                ActionRequest::Attack { attacker_id, defender_id } => {
                    Action::Attack { attacker_id, defender_id }
                }
                ActionRequest::EndTurn => Action::EndTurn,
            }
        }
    }
    ```

    /// Serializes the current game state as a JSON string.
    /// Returns an empty string if no game has been created or serialization fails.
    #[func]
    fn get_state_json(&self) -> GString {
        let Some(state) = self.game.as_ref() else { return "".into() };
        match serde_json::to_string(&StateSnapshot::from_game_state(state)) {
            Ok(s) => s.into(),
            Err(e) => {
                godot_error!("get_state_json: serialization failed: {}", e);
                "".into()
            }
        }
    }

    /// Parse a JSON action string and apply it to the game.
    /// Returns 0 on success, -99 on JSON parse error, or a negative ActionError code.
    ///
    /// Expected formats:
    ///   {"action":"Move","unit_id":1,"col":3,"row":2}
    ///   {"action":"Attack","attacker_id":1,"defender_id":2}
    ///   {"action":"EndTurn"}
    #[func]
    fn apply_action_json(&mut self, json: GString) -> i32 {
        let Some(state) = self.game.as_mut() else { return -1 };
        let req: ActionRequest = match serde_json::from_str(&json.to_string()) {
            Ok(r) => r,
            Err(e) => {
                godot_error!("apply_action_json: parse error: {}", e);
                return -99;
            }
        };
        match apply_action(state, req.into()) {
            Ok(()) => 0,
            Err(e) => action_err_code(e),
        }
    }
    ```
