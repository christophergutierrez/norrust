<objective>
## Goal
Implement the advancement action end-to-end: a pure Rust `advance_unit()` function,
an `apply_advance()` GDExtension bridge method for GDScript, `ActionRequest::Advance`
for the JSON AI API, and a headless simulation test that exercises the full
XP-gain → advancement-pending → advance chain.

## Purpose
With XP accumulating correctly (08-01), the game now needs a way to actually advance
units. Phase 9 will surface advancement in the UI — it needs `apply_advance()` to
already exist. The simulation test provides balance confidence: we can run many
combats headlessly and verify that the XP threshold and stat upgrades work as expected
before any visual work begins.

## Output
- `unit.rs`: `pub fn advance_unit(unit: &mut Unit, new_def: &UnitDef)` — stat mutation,
  xp reset, advancement_pending cleared, def_id updated
- `gdext_node.rs`: `apply_advance(unit_id: i32) -> i32` — registry lookup, validation,
  calls `advance_unit()`, returns error codes
- `snapshot.rs`: `ActionRequest::Advance { unit_id: u32 }` variant + dispatch in
  `apply_action_json()` (routes to `apply_advance` logic)
- `tests/simulation.rs`: `test_headless_advancement_scenario` — fights until
  advancement_pending, advances, asserts new stats
</objective>

    ```rust
    /// Advance `unit` to the stats defined by `new_def`.
    ///
    /// Copies all stat fields from `new_def` (hp/max_hp → new_def.max_hp, movement,
    /// movement_costs, attacks, defense, resistances, xp_needed from new_def.experience).
    /// Resets xp to 0, clears advancement_pending, and updates def_id.
    pub fn advance_unit(unit: &mut Unit, new_def: &UnitDef) {
        unit.def_id = new_def.id.clone();
        unit.max_hp = new_def.max_hp;
        unit.hp = new_def.max_hp;           // full heal on advancement
        unit.movement = new_def.movement;
        unit.movement_costs = new_def.movement_costs.clone();
        unit.attacks = new_def.attacks.clone();
        unit.defense = new_def.defense.clone();
        unit.resistances = new_def.resistances.clone();
        unit.xp_needed = new_def.experience;
        unit.xp = 0;
        unit.advancement_pending = false;
    }
    ```

    Add `use crate::unit::advance_unit;` at the top import section.

    Add `apply_advance` as a new `#[func]` in the `impl NorRustCore` block, after
    `apply_attack`:

    ```rust
    /// Advance a unit to its next form. The unit must belong to the active faction
    /// and have advancement_pending = true.
    ///
    /// Returns:
    ///   0  = success
    ///  -1  = unit not found
    ///  -2  = not your turn (unit belongs to wrong faction)
    ///  -8  = unit not ready to advance (advancement_pending is false)
    ///  -9  = no advancement target (advances_to is empty in UnitDef)
    /// -10  = target UnitDef not found in registry
    #[func]
    fn apply_advance(&mut self, unit_id: i32) -> i32 {
        let Some(state) = self.game.as_mut() else { return -1 };
        let uid = unit_id as u32;

        // Validate unit exists and belongs to active faction
        let (current_def_id, advancement_pending, faction) = match state.units.get(&uid) {
            Some(u) => (u.def_id.clone(), u.advancement_pending, u.faction),
            None => return -1,
        };
        if faction != state.active_faction {
            return -2;
        }
        if !advancement_pending {
            return -8;
        }

        // Look up advances_to from registry
        let target_def_id = match self.units.as_ref()
            .and_then(|r| r.get(&current_def_id))
            .and_then(|def| def.advances_to.first().cloned())
        {
            Some(id) => id,
            None => return -9,
        };

        // Clone the target UnitDef before mutably borrowing state
        let new_def = match self.units.as_ref().and_then(|r| r.get(&target_def_id)) {
            Some(def) => def.clone(),
            None => return -10,
        };

        // Apply advancement
        let unit = state.units.get_mut(&uid).unwrap();
        advance_unit(unit, &new_def);
        0
    }
    ```

    ```rust
    #[test]
    fn test_headless_advancement_scenario() {
        // ── Setup: fighter with xp_needed=40 vs. repeated weak enemies ──
        let board = Board::new(5, 5);
        let mut state = GameState::new_seeded(board, 7);

        let sword = AttackDef {
            id: "sword".to_string(), name: "Sword".to_string(),
            damage: 100, strikes: 1,
            attack_type: "blade".to_string(), range: "melee".to_string(),
        };
        let mut fighter = Unit::new(1, "fighter", 30, 0);
        fighter.attacks = vec![sword.clone()];
        fighter.xp_needed = 40;
        state.place_unit(fighter, Hex::from_offset(0, 0));

        // Kill enough enemies to accumulate 40+ XP (each kill = 9 XP: 1 hit + 8 kill)
        // 5 kills × 9 XP = 45 XP → advancement_pending
        for enemy_id in 2u32..=6 {
            // Place a fresh 1-HP enemy adjacent at (1,0)
            let mut enemy = Unit::new(enemy_id, "enemy", 1, 1);
            enemy.default_defense = 0;
            state.place_unit(enemy, Hex::from_offset(1, 0));

            apply_action(
                &mut state,
                Action::Attack { attacker_id: 1, defender_id: enemy_id },
            )
            .expect("attack should succeed");

            // EndTurn between kills to reset attacked flag
            apply_action(&mut state, Action::EndTurn).expect("EndTurn should succeed");
            // EndTurn again to return to faction 0
            apply_action(&mut state, Action::EndTurn).expect("EndTurn should succeed");
        }

        assert!(state.units[&1].xp >= 40, "fighter should have accumulated 40+ XP");
        assert!(state.units[&1].advancement_pending, "advancement_pending must be set");

        // ── Advance the fighter to hero ──────────────────────────────────
        let hero_def = UnitDef {
            id: "hero".to_string(),
            name: "Hero".to_string(),
            max_hp: 45,
            movement: 5,
            attacks: vec![sword],
            resistances: HashMap::new(),
            movement_costs: HashMap::new(),
            defense: HashMap::new(),
            level: 2,
            experience: 80,
            advances_to: vec![],
        };

        advance_unit(state.units.get_mut(&1).unwrap(), &hero_def);

        assert_eq!(state.units[&1].def_id, "hero");
        assert_eq!(state.units[&1].max_hp, 45);
        assert_eq!(state.units[&1].hp, 45);
        assert_eq!(state.units[&1].xp, 0);
        assert!(!state.units[&1].advancement_pending);
    }
    ```
